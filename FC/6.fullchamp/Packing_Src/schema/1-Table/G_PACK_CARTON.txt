DROP TABLE SAJET.G_PACK_CARTON CASCADE CONSTRAINTS;

CREATE TABLE SAJET.G_PACK_CARTON
(
  WORK_ORDER     VARCHAR2(25 BYTE),
  PART_ID        NUMBER,
  CARTON_NO      VARCHAR2(30 BYTE),
  CLOSE_FLAG     VARCHAR2(8 BYTE),
  PROCESS_ID     NUMBER,
  CREATE_EMP_ID  NUMBER,
  CREATE_TIME    DATE                           DEFAULT SYSDATE,
  PKSPEC_ID      NUMBER,
  CLOSE_TIME     DATE,
  CLOSE_EMP_ID   NUMBER,
  LOT_SEQ        NUMBER,
  LOT_NO         VARCHAR2(30 BYTE),
  LEVEL_CODE     VARCHAR2(5 BYTE)
)
TABLESPACE SYSRT
RESULT_CACHE (MODE DEFAULT)
LOGGING 
NOCOMPRESS 
NOCACHE
NOPARALLEL
MONITORING;

COMMENT ON TABLE SAJET.G_PACK_CARTON IS '包裝站包裝箱號狀態表';

COMMENT ON COLUMN SAJET.G_PACK_CARTON.WORK_ORDER IS '工單號碼';

COMMENT ON COLUMN SAJET.G_PACK_CARTON.PART_ID IS '機種名稱';

COMMENT ON COLUMN SAJET.G_PACK_CARTON.CARTON_NO IS '箱號';

COMMENT ON COLUMN SAJET.G_PACK_CARTON.CLOSE_FLAG IS '裝箱狀況(N:未滿,Y:已滿)';

COMMENT ON COLUMN SAJET.G_PACK_CARTON.PROCESS_ID IS '站別';

COMMENT ON COLUMN SAJET.G_PACK_CARTON.CREATE_EMP_ID IS '建立人員';

COMMENT ON COLUMN SAJET.G_PACK_CARTON.CREATE_TIME IS '建立時間';

COMMENT ON COLUMN SAJET.G_PACK_CARTON.LOT_SEQ IS '箱内序号';

COMMENT ON COLUMN SAJET.G_PACK_CARTON.LOT_NO IS '批次号';

COMMENT ON COLUMN SAJET.G_PACK_CARTON.LEVEL_CODE IS '等级';



CREATE INDEX SAJET.G_PACK_CARTON_CARTON ON SAJET.G_PACK_CARTON
(CARTON_NO)
LOGGING
TABLESPACE SYSRTIDX
NOPARALLEL;


CREATE INDEX SAJET.G_PACK_CARTON_MODEL_IDX ON SAJET.G_PACK_CARTON
(PART_ID)
LOGGING
TABLESPACE SYSRTIDX
NOPARALLEL;
