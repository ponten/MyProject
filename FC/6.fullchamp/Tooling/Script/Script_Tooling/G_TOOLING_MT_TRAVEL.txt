CREATE TABLE SAJET.G_TOOLING_MT_TRAVEL
(
  TOOLING_ID     NUMBER,
  STATUS         VARCHAR2(5 BYTE)               DEFAULT 'T',
  UPDATE_USERID  NUMBER,
  UPDATE_TIME    DATE                           DEFAULT SYSDATE,
  MEMO           VARCHAR2(500 BYTE),
  USED_COUNT     NUMBER,
  RESULT         VARCHAR2(5 BYTE)
)
TABLESPACE SYSRT
RESULT_CACHE (MODE DEFAULT)
LOGGING 
NOCOMPRESS 
NOCACHE
NOPARALLEL
MONITORING;

COMMENT ON TABLE SAJET.G_TOOLING_MT_TRAVEL IS 'Tooling狀態歷程記錄表(領用,歸還,保養)';

COMMENT ON COLUMN SAJET.G_TOOLING_MT_TRAVEL.TOOLING_ID IS 'SYS_TOOLING.TOOLING_ID';

COMMENT ON COLUMN SAJET.G_TOOLING_MT_TRAVEL.STATUS IS '狀態(領用,歸還,保養)';

COMMENT ON COLUMN SAJET.G_TOOLING_MT_TRAVEL.UPDATE_USERID IS '資料異動人員';

COMMENT ON COLUMN SAJET.G_TOOLING_MT_TRAVEL.UPDATE_TIME IS '資料異動時間(此次狀態時間)';

COMMENT ON COLUMN SAJET.G_TOOLING_MT_TRAVEL.MEMO IS '備註';

COMMENT ON COLUMN SAJET.G_TOOLING_MT_TRAVEL.USED_COUNT IS '使用次數';

COMMENT ON COLUMN SAJET.G_TOOLING_MT_TRAVEL.RESULT IS '驗收結果(OK,NG)';
